# 비만도 계산

> 날짜: 2026-01-28
> 원본 노션: [링크](https://www.notion.so/2f7b28703eb0802ea258ebe3e583bc3a)

---

(키 - 100) * 0.9 : 표준 
비만도 : (몸무게 / 표준몸무게) * 100
비만도 > 120 : 비만 
비만도 < 80 : 저체중 
80 < 정상 < 120

```java
package gui;

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.text.NumberFormat;

import javax.swing.*;
import javax.swing.text.AbstractDocument;
import javax.swing.text.AttributeSet;
import javax.swing.text.BadLocationException;
import javax.swing.text.DocumentFilter;
import javax.swing.text.DocumentFilter.FilterBypass;
import javax.swing.text.NumberFormatter;

// 비만도 계산하는 프로그램

public class BMD extends JFrame{
	private JTextField nameTF, heightTF, weightTF;
	private JButton resultBtn, deleteBtn;
	private JLabel resultLb;
	
	public BMD() {
		setLayout(new GridLayout(5,1));
		
		//이름
		JPanel nameP = new JPanel();
		nameP.setLayout(new GridLayout(1,2));
		nameP.add(new JLabel("이름"));
		nameTF = new JTextField();
		nameP.add(nameTF);
		add(nameP);
		
		//키
		JPanel heightP = new JPanel();
		heightP.setLayout(new GridLayout(1,2));
		heightP.add(new JLabel("키"));
		heightTF = new JTextField();
		
		
		//숫자, 실수만 입력받게 수정 방법1(이 방법을 주로 사용)
		((AbstractDocument)heightTF.getDocument()).setDocumentFilter(
	            new DocumentFilter(){
	         @Override
	         public void replace(FilterBypass fb, int offset, int length, String text, AttributeSet attrs)
	               throws BadLocationException {
	            if(text.matches("\\d*\\.?\\d*")) {
	               super.replace(fb, offset, length, text, attrs);
	            } else {
	            	//숫자가 아닌 경우 경고음 나게 하기 
	            	Toolkit.getDefaultToolkit().beep();
	            }
	         }
	      }); //숫자, 실수만 입력받게 수정 닫기 
		
		
		heightP.add(heightTF);
		add(heightP);
		
		//몸무게
		JPanel weightP = new JPanel();
		weightP.setLayout(new GridLayout(1,2));
		weightP.add(new JLabel("몸무게"));
		weightTF = new JTextField();
		
		
		
		//숫자, 실수만 입력받게 수정 방법2
		NumberFormat format = NumberFormat.getInstance();
		format.setGroupingUsed(false);
		NumberFormatter formatter = new NumberFormatter(format);
		formatter.setValueClass(Double.class);
		formatter.setMinimum(0.0);				//몸무게 최소값
		formatter.setMaximum(Double.MAX_VALUE);	//몸무게 최대값
		formatter.setAllowsInvalid(false);		//이외 영문자 특수문자 등 차단
		JFormattedTextField weightTF2 = new JFormattedTextField(formatter);
		weightP.add(weightTF2);
		//숫자, 실수만 입력받게 수정 닫기 
		
		//weightP.add(weightTF);
		add(weightP);
		
		//버튼
		JPanel buttonP = new JPanel();
		buttonP.setLayout(new GridLayout(1,2));
		resultBtn = new JButton("결과보기");
		buttonP.add(resultBtn);
		deleteBtn = new JButton("초기화");
		buttonP.add(deleteBtn);
		add(buttonP);
	
		//결과 
		resultLb = new JLabel("결과가 여기에 표시됩니다.");
		add(resultLb);
		
		//초기화 버튼을 누르면 : TF에 내용을 모두 지우기 = setText();
		deleteBtn.addActionListener(new ActionListener() {	
			@Override
			public void actionPerformed(ActionEvent e) {
				nameTF.setText("");
				heightTF.setText("");
				//weightTF.setText("");
				weightTF2.setText("");
				
			}
		});
		
		
		//결과보기 
		resultBtn.addActionListener(new ActionListener() {
			
			@Override
			public void actionPerformed(ActionEvent e) {
				
//				String name = nameTF.getText(); 
//				double weigth = Double.parseDouble(weightTF.getText());
//				double heigth = Double.parseDouble(heightTF.getText());
//				double stnWeigth, bmi;
				String name;
				double heigth,weigth,stnWeigth, bmi;
				
				try {
					if(nameTF.getText().equals("")) {
						JOptionPane.showMessageDialog(nameP, "이름을 입력하세요");
					}else if(heightTF.getText().equals("")){
						JOptionPane.showMessageDialog(heightP, "키를 입력하세요");
					}else if(weightTF2.getText().equals("")){
						JOptionPane.showMessageDialog(weightP, "몸무게를 입력하세요");
					}else {
						name = nameTF.getText();
						heigth = Double.parseDouble(heightTF.getText());
						weigth = Double.parseDouble(weightTF2.getText());
						
						
						stnWeigth = (heigth - 100) * 0.9;
						bmi = (weigth / stnWeigth) * 100;
						
						if (bmi>120) {
							resultLb.setText(name + "님은 비만입니다.");
						}else if (bmi<80) {
							resultLb.setText(name + "님은 저체중입니다.");
						}else{
							resultLb.setText(name + "님은 정상입니다.");
						}
					}
				} catch (NumberFormatException e1) {
					JOptionPane.showMessageDialog(null, "키와 몸무게는 숫자로 작성해 주세요");
				}
			}
		});

	}
	public static void main(String[] args) {
		BMD bmd = new BMD();
		bmd.setSize(300,600);
		bmd.setVisible(true);
		bmd.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		
	}
}
```
